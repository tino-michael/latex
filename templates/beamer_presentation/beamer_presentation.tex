\documentclass[8pt]{beamer}

% all of these options are optional though some would make less sense to omit in most cases...
\usepackage[
    title={my title},
    author={my name},
    meeting={my meeting},
    subtitle={my subtitle},
    shorttitle={my shorttitle}, % defaults to `title`
    institute={my institute},
    shortinstitute={my short institute},
    date={1970-01-01} % defaults to `\isodate`
    ]{beamersetup}


% to show multiple logos in a `titlegraphic` put them within a `tikzpicture`
\titlegraphic
{
    \begin{tikzpicture}
        \node (centre) at (0,0)
        { \includegraphics[height=40pt]{300.300.ff0.png} };
        \node[anchor=west] at ([xshift=-150pt]centre)
        { \includegraphics[height=40pt]{200.200.f0f.png} };
        \node[anchor=east] at ([xshift=+150pt]centre)
        { \includegraphics[height=40pt]{400.400.0ff.png} };
    \end{tikzpicture}
}

\begin{document}

    \begin{frame}[noframenumbering,plain]
        \titlepage
    \end{frame}

    \begin{frame}{First slide}
        \begin{itemize}
            \item first item
            \item second item
        \end{itemize}
    \end{frame}

    % put backup slides inside the `backup` command to be able to quickly hide them all
    \backup[hide]
    {
        \begin{frame}
            does not shown backup
        \end{frame}
    }
    \backup
    {
        \begin{frame}
            shows backup
        \end{frame}
    }
\end{document}
